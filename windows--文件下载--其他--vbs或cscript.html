var xmlHTTP = new ActiveXObject("Microsoft.XMLHTTP");
xmlHTTP.open("Get", "https://www.baidu.com", false);
xmlHTTP.send();
s=new ActiveXObject("ADODB.Stream");
s.Type=1;
s.Open();
s.Write(xmlHTTP.responseBody);
s.SaveToFile("C:\\Users\\xiaohao\\Desktop\\2.txt",2);
